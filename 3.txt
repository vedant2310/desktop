Interest Procedure:
CREATE OR REPLACE PROCEDURE calc_int(p_acc NUMBER, p_rate NUMBER) IS
    v_bal NUMBER; v_yr NUMBER;
BEGIN
    SELECT Balance, NoOfYears INTO v_bal, v_yr FROM Account WHERE Account_No = p_acc;
    INSERT INTO Earned_Interest VALUES (p_acc, (v_bal*v_yr*p_rate)/100);
END;

Function (>50k):
CREATE OR REPLACE FUNCTION get_rich_acc RETURN SYS_REFCURSOR IS
    rc SYS_REFCURSOR;
BEGIN
    OPEN rc FOR SELECT * FROM Account WHERE Balance > 50000;
    RETURN rc;
END;
